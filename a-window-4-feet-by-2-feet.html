[tex]\sf A = \frac{(b_1 + b_2)}{2} \times h\\\\where,\\b_1 = length \;of \;base \;1 = 92 \;m\\b_2 = length \;of \;base \;2 = 128 \;m\\h = height = 40 \;cm\\\\solve \;for \;A,\\\sf A = \frac{(92 \;+ \;128)}{2} \times 40\\\\\sf A = \frac{(220)}{2} \times 40\\\\\sf A = 110 \times 40\\\\\sf A = 4,400 \;m^{2}[/tex]
