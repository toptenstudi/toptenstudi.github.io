[tex]c = 826 - 273[/tex]

[tex]c = 553[/tex]

Final Answer:

553°C

3. 72 degree celsius to Fahrenheit

[tex]c = 72[/tex]

[tex]f = 32 + \frac{9}{5} \times 72[/tex]

[tex]f = 32 + \frac{648}{5} [/tex]

[tex]f = 32 + 129 \: \frac{3}{5} [/tex]

[tex]f = 161 \: \frac{3}{5} [/tex]

[tex]f = 161.6[/tex]

Final Answer:

161.6°F

[tex]c = 164[/tex]

[tex]k = 164 + 273.15[/tex]

[tex]k = 437.15[/tex]

Final Answer:

427.15 K

[tex]k = 18[/tex]

[tex]f = 1.8(18 - 273) + 32[/tex]
